/* creating tcga_master_table from tcga_master view select into statement */

create table tcga_master_table 
   select sample_id, patient_id, hugo_symbol, gene, transcript_id, cancer_type
   from tcga_master_view;

/* example query for mining tcga master table */

select sample_id, cancer_type, hugo_symbol from tcga_master_table where hugo_symbol = "CLN3" or hugo_symbol = "SMOC1" or hugo_symbol = "TP53" order by hugo_symbol asc;

select * from tcga_master_table
   group by patient_id 
   having count(patient_id) > 50 and
   count(hugo_symbol) > 15;

/* Joining genetpm1 table from GTex to TCGA_Master_table */
